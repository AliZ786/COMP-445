GET STATEMENTS
---------------------
python httpc.py get -u "http://httpbin.org/get?course=networking&assignment=1"
python httpc.py get -v -u "http://httpbin.org/get?course=networking&assignment=1"
python httpc.py get -v -h Content-Type:application/json -u "http://httpbin.org/get?course=networking&assignment=1"
python httpc.py get -v -h Content-Type:application/json -h  hello:bello -u "http://httpbin.org/get?course=networking&assignment=1"
python httpc.py get -v 'http://httpbin.org/get?course=networking&assignment=1' -o hello.txt



POST STATEMENTS
---------------------
python httpc.py post "http://httpbin.org/post"
python httpc.py post -v  "http://httpbin.org/post"
python httpc.py post -h Content-Type:application/json "http://httpbin.org/post"
python httpc.py post -v -h Content-Type:application/json  "http://httpbin.org/post"
python httpc.py post -v --d '{"Assignment": 1}' "http://httpbin.org/post"
python httpc.py post -v -h Content-Type:application/json -f test.txt "http://httpbin.org/post"
python httpc.py post -v -h Content-Type:application/json "http://httpbin.org/post" -o output_post.txt

---------------------
